# 数组去重

  • 方法一:indexOf
    let arr = [1, 2, 3, 1, 3, 5, 5, 7, 8]
    function unique(params) {
      let newArr = []
      params.forEach(item => {
        if (newArr.indexOf(item) === -1) {
          newArr.push(item)
        }
      })
      return newArr
    }
    unique(arr)
    console.log(unique(arr))// [1, 2, 3, 5, 7, 8]
  • 方法二:includes
    let arr = [1, 2, 3, 1, 3, 5, 5, 7, 8]
    function unique(params) {
      let newArr = []
       params.forEach(item => {
        if(!newArr.includes(item)){
          newArr.push(item)
        }
      })
      return newArr
    }
    unique(arr)
    console.log(unique(arr))// [1, 2, 3, 5, 7, 8]
  • 方法三:filter+indexOf通过过滤那些元素第一次出现的元素来实现去重
    let arr = [1, 2, 3, 1, 3, 5, 5, 7, 8]
 function unique(params) {
   return params.filter((item, index) => {
     return params.indexOf(item) === index
   })
 }
 unique(arr)
 console.log(unique(arr))// [1, 2, 3, 5, 7, 8]


- 方法四:sort()排序,相邻两个元素不相等
    ```js
    let arr = [1, 2, 3, 1, 3, 5, 5, 7, 8]
    function unique(params) {
      let newArr = []
      params=params.sort()
      params.forEach((item,index) => {
        if(params[index]!==params[index+1]){
          newArr.push(item)
        }
      });
      return newArr;
    }
    unique(arr)
    console.log(unique(arr))// [1, 2, 3, 5, 7, 8]
  • 方法五:双重循环去重实现。
    let arr = [1, 2, 3, 1, 3, 5, 5, 7, 8]
    function unique(params) {
      let newArr = [],
        isRepeat;
      for (let i = 0; i < params.length; i++) {
        isRepeat = false;
        for (let j = 0; j < newArr.length; j++) {
          if (params[i] === newArr[j]) {
            isRepeat = true
            break;
          }
        }
        if (!isRepeat) {
          newArr.push(params[i]);
        }
      }
      return newArr;
    }
    unique(arr)
    console.log(unique(arr))// [1, 2, 3, 5, 7, 8]
  • 方法六:reduce实现去重
    let arr = [1, 2, 3, 1, 3, 5, 5, 7, 8]
    function unique(params) {
      return params.reduce((pre, next) => {
        if (!pre.includes(next)) {
          pre.push(next)
        }
        return pre
      }, [])
    }
    unique(arr)
    console.log(unique(arr))// [1, 2, 3, 5, 7, 8]
  • 方法七:new Set()实现去重
let arr = [1, 2, 3, 1, 3, 5, 5, 7, 8]
   function unique(params) {
      return Array.from(new Set(params)) 
    }
    unique(arr)
    console.log(unique(arr))// [1, 2, 3, 5, 7, 8]
